CREATE TABLE jobs(
    job_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    job_title VARCHAR(255),
    min_salary DECIMAL(10),
    max_salary DECIMAL(10),
    PRIMARY KEY(job_id)
);
CREATE TABLE job_history(
    start_date DATE,
    end_date DATE,
    max_salary DECIMAL(10)
    employee_id number(2) REFERENCES employees(employee_id),
);
CREATE TABLE departments(
    department_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    department_name VARCHAR(255),
    PRIMARY KEY(department_id)
);
CREATE TABLE employees(
    employee_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    first_name VARCHAR(255),
    last_name VARCHAR(255),
    email VARCHAR(255),
    phone_number VARCHAR(12),
    hire_date DATE,
    salary DECIMAL(10),
    commision_pct VARCHAR(255),
    PRIMARY KEY(employee_id)
);
CREATE TABLE locations(
    location_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    street_address VARCHAR(255),
    postal_code VARCHAR(6),
    emacityil VARCHAR(255),
    state_province VARCHAR(12),
    PRIMARY KEY(location_id)
);
CREATE TABLE countries(
    country_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    country_name VARCHAR(255),
    PRIMARY KEY(country_id)
);
CREATE TABLE regions(
    region_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    region_name VARCHAR(255),
    PRIMARY KEY(region_id)
)
